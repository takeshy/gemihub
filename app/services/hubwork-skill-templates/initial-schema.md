## accounts
- email: string (e.g. "jane@example.com")
- name: string (e.g. "Jane Doe")
- created_at: datetime (e.g. "2025-04-01T10:00:00Z")
- logined_at: datetime (e.g. "2025-04-10T09:30:00Z")

## meetings
- id: string (e.g. "uuid")
- user_email: string (e.g. "jane@example.com")
- title: string (e.g. "Jane Doe - Preliminary Booking")
- scheduled_at: datetime (e.g. "2025-04-15T10:00:00Z")
- status: string (e.g. "booked")
- created_at: datetime (e.g. "2025-04-12T09:00:00Z")
